react组件

  1. 组件的首字母必须大写
  2. return返回的view必须是HTMLview
  3. 在渲染的时候只能存在一个根标签
  4. 在组件中标签必须闭合，否则报错
  5. 组件与组件的嵌套
  6. style样式的添加class需要改成className

props

//声明组件hello
 var Hello = React.createClass({
   render:function () {
   //通过props获得组件内属性title的值 进行判断
       var value = this.props.title;
       if (value == 'Hello'){
           value = 'Hello';
       }else {
           value = '我不是Hello'
       }
       返回value值
       return <p>{value}</p>
   }
});
//声明组件Jean
var Jean = React.createClass({
//对传入的数据做限制
   propTypes:{
   //限制为字符串
       name:React.PropTypes.string,
       //限制为number 且不为空
       age:React.PropTypes.number.isRequired
   },
   //对参数做初始化
   getDefaultProps:function () {
       return {
           name:'jean'
       }
   },
   //组件的视图
   render:function () {
       return (
               <div>
                   <p>{this.props.name}</p>
                   <p>{this.props.age}</p>
               </div>
       )
   }
});
ReactDOM.render(
    <div>
        <Jean age={20} />
        <Hello title="Hello1"/>
    </div>,
    document.getElementById('root')
)

事件

var MyEvent = React.createClass({
//声明事件
   handerClick:function () {
       alert('HAHA');
   },
   //通过refs获取指定input内的值
   handerGetContent:function () {
      console.log(this.refs.inputref.value) ;
   },
   render:function () {
       return (
           <div>
           //给input添加ref属性 以便获取input内的值
               <input type="text" ref="inputref"/>
               //绑定事件没有onclick后面不能添加引号
               <button onClick={this.handerGetContent}>获取输入框的内容</button>
               <button onClick={this.handerClick}>按钮</button>
           </div>
       )
   }
});
ReactDOM.render(
      <MyEvent />  ,
    document.getElementById('root')
)

状态state

  var MyState = React.createClass({
  //初始化状态值
    getInitialState:function(){
        return{
            content:true
        }
    },
    //通过事件修改状态值
    handlerClick:function(){
        this.setState({
            content: !this.state.content
        })
    },
    render:function(){
    //声明变量flag等于状态值
        var flag = this.state.content;
        return(
                <div>
                    <p>
                        {
                        //根据状态值修改视图
                            flag ? "我是孙悟空" : "我是通臂猿猴"
                        }
                    </p>
                    //元素绑定点击事件
                    <button onClick={this.handlerClick}>出来吧，皮卡丘</button>
                </div>
        )
    }
});
ReactDOM.render(
        <MyState />,
    document.getElementById("root")
)

组件的生命周期

三个阶段：

  1. Mounting：组件的渲染阶段
  2. Updating：组件的更新阶段
  3. Unmounting：组件的卸载阶段

    react分别为这三个状态提供了will和did两个函数

    1. componentWillMount() 组件渲染前触发事件
    2. componentDidMount() 组件渲染后触发事件
    3. componentWillUpdate() 组件更新前触发事件
    4. componentDidUpdate() 组件更新后触发事件
    5. componentWillUnmount() 组件卸载前触发事件
    6. 提供两个额外的生命周期函数
    7. shouldComponentUpdate()
    8. componentWillReceiveProps()
var MyComponent = React.createClass({
    //初始化传入的参数
    getDefaultPoprs:function(){
      return{
        title:"嘿嘿"
      }
    },
    //初始化状态
    getInitialState:function(){
      return {
        mystate:"初始化state"
      }
    },
    componentWillMount:function(){
      console.log("组件渲染前");
    },
    componentDidMount:function(){
      console.log("组件渲染后");
    },
    componentWillUpdate:function(){
      console.log("组件更新前");
    },
    componentDidUpdate:function(){
      console.log("组件更新后");
    },
    componentWillReceiveProps:function(){
      console.log("属性发生改变");
    },
    shouldComponentUpdate:function(nextProps, nextState){
       return{

       }
    },
    componentWillUnmount:function(){
      console.log("组件卸载前");
    },

    handlerClick:function(){
      this.setState({
        mystate:"我是改变之后的状态"
      })
    },

    handlerSytle:function(){

    },

        color:"green",
        fontSize:"30px",

生命周期结合ajax

使用ajax时注意修改this指向

生命周期图

//声明组件
var MyComponent = React.createClass({
//初始化参数
    getDefaultPoprs:function () {
        return{
            url:''
        }
    },
    //初始化状态
    getInitialState:function () {
        return {
            imgurl:''
        }
    },
    //ajax应该在组件渲染完成后进行
    componentDidMount:function () {
        $.ajax({
            type:'get',
            url:this.props.url,
            success:function (data) {
                console.log(1);
                this.setState({
                    imgurl:data.banner[0].img
                })
            }.bind(this)  //修改this指向
        })
    },
    handlerClick:function () {
        this.setState({
            imgurl:'http://iwen.wiki/sxtstu/blueberrypai/indexImg/banner2.jpg'
        })
    },
    render:function () {
        return(
            <div>
                <img src={this.state.imgurl} alt=""/>
                <button onClick={this.handlerClick}>更换图片</button>
            </div>
        )
    }
});
ReactDOM.render(
//传入请求地址
    <MyComponent url="http://www.iwen.wiki/sxtstu/blueberrypai/getIndexBanner.php"/> ,
    document.getElementById('root')
)
